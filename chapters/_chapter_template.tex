\begin{workedexample}
\begin{example}
Evaluate \(\lim_{x\to a^-}f(x)\), \(\lim_{x\to a^+}f(x)\), and determine
whether \(\lim_{x\to a}f(x)\) exists.
\end{example}

\begin{solution}
If \(x\to a^{-}\), then the relevant branch or sign information gives
\(\lim_{x\to a^-}f(x)=L_-\).

If \(x\to a^{+}\), then the relevant branch or sign information gives
\(\lim_{x\to a^+}f(x)=L_+\).

Since \(L_- \ne L_+\), \(\lim_{x\to a}f(x)\) does not exist.
\end{solution}
\end{workedexample}

\begin{strategy}
Use a \texttt{strategy} env when several worked examples share a common
method. Spell the method out as a short numbered list so the reader can
refer back to it as a checklist.
\begin{enumerate}
    \item First step of the method.
    \item Second step, with any branching conditions.
    \item Third step, and what to do if the first two fail.
\end{enumerate}
\end{strategy}

\begin{workedexample}
\begin{example}
Evaluate
\[
\lim_{x\to a}\frac{\sqrt{x+b}-c}{x-a}.
\]
\end{example}

\begin{solution}
Direct substitution gives the indeterminate form \(\frac{0}{0}\), so we
rationalize:
\[
\frac{\sqrt{x+b}-c}{x-a}
\cdot
\frac{\sqrt{x+b}+c}{\sqrt{x+b}+c}
=
\frac{x+b-c^2}{(x-a)\bigl(\sqrt{x+b}+c\bigr)}.
\]
After simplification, we obtain
\[
\frac{1}{\sqrt{x+b}+c}.
\]
Therefore,
\[
\lim_{x\to a}\frac{\sqrt{x+b}-c}{x-a}
=
\frac{1}{2c}.
\]
\end{solution}
\end{workedexample}

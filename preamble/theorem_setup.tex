% Reserve more space for formal result blocks and less for worked material that may flow naturally.
\newcommand{\formalblockneedspace}{8\baselineskip}
\newcommand{\supportingblockneedspace}{7\baselineskip}
\newcommand{\exampleblockneedspace}{6\baselineskip}
\newcommand{\exerciseblockneedspace}{4\baselineskip}
\newcommand{\solutionblockneedspace}{5\baselineskip}
\newcommand{\proofblockneedspace}{3\baselineskip}
% All block-level "reserve space first" checks funnel through this wrapper.
% In normal typesetting it uses Needspace's exact page-remaining test. During
% workedexample measurement we temporarily suppress it so the measured vbox
% reflects only real content height, not extra penalties/vfil from nested
% needspace checks.
\newif\ifmeasuringworkedexample
\newcommand{\blockneedspace}[1]{%
  \ifmeasuringworkedexample\else\Needspace{#1}\fi
}
\newcommand{\reserveblockspace}[2]{%
  \AddToHook{env/#1/before}{\blockneedspace{#2}}%
}

% ---------------------------------------------------------------------------
% Formal-statement environments.
% Per CONTENT_README.md v3.0 (Numbering and Cross-References), each formal
% environment has its own chapter-scoped counter. This is a change from the
% v2.x shared-counter pattern: high-school self-study readers looking up
% "Definition 1.3" expect it to be the third definition, not the third
% formal statement of any type. Per-env counters match that expectation.
%
% Note: `lemma` is intentionally absent. In the HS audience, the cognitive
% cost of distinguishing lemma from theorem outweighs the benefit; results
% that would be lemmas in a graduate-level text are either absorbed into
% proofs or promoted to proposition.
% ---------------------------------------------------------------------------

\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[chapter]
\newtheorem{proposition}{Proposition}[chapter]
\newtheorem{corollary}{Corollary}[chapter]

\theoremstyle{definition}
\newtheorem{definition}{Definition}[chapter]

% Worked blocks keep their own chapter-scoped counters.
\newtheorem{example}{Example}[chapter]
\newtheorem{exercise}{Exercise}[chapter]

\theoremstyle{remark}
\newtheorem{remark}{Remark}[chapter]

% ---------------------------------------------------------------------------
% `caution` and `strategy` environments.
% Added in v3.0 to support the Stewart-style self-study register. Visually
% distinct from `remark` via a left-side coloured accent bar, implemented
% with mdframed so page breaks still work for strategy boxes that contain
% multi-step numbered lists.
%
%   caution  -- red bar, used for notation traps, common errors, restrictions
%               that are easy to miss. Typically 1-3 sentences.
%   strategy -- blue bar, used for Stewart-style problem-solving method
%               boxes (numbered step list or short paragraph).
%
% Both carry chapter-scoped counters of their own. Labels are optional (see
% CONTENT_README.md §6).
% ---------------------------------------------------------------------------
\mdfdefinestyle{cautionmdstyle}{%
  leftline=true, rightline=false, topline=false, bottomline=false,
  linewidth=2pt, linecolor=colorcaution,
  innerleftmargin=0.9em, innerrightmargin=0em,
  innertopmargin=0.25em, innerbottommargin=0.25em,
  skipabove=0.6em, skipbelow=0.6em,
  nobreak=false,
}
\mdfdefinestyle{strategymdstyle}{%
  leftline=true, rightline=false, topline=false, bottomline=false,
  linewidth=2pt, linecolor=colorprimary,
  innerleftmargin=0.9em, innerrightmargin=0em,
  innertopmargin=0.25em, innerbottommargin=0.25em,
  skipabove=0.6em, skipbelow=0.6em,
  nobreak=false,
}

\theoremstyle{remark}
\newmdtheoremenv[style=cautionmdstyle]{caution}{Caution}[chapter]
\newmdtheoremenv[style=strategymdstyle]{strategy}{Strategy}[chapter]

% Pagination protection.
% Formal result blocks reserve more page-bottom space; worked material and
% aside environments use lighter protection.
% ---------------------------------------------------------------------------
\reserveblockspace{theorem}{\formalblockneedspace}
\reserveblockspace{proposition}{\formalblockneedspace}
\reserveblockspace{corollary}{\formalblockneedspace}
\reserveblockspace{definition}{\formalblockneedspace}
\reserveblockspace{example}{\exampleblockneedspace}
\reserveblockspace{exercise}{\exerciseblockneedspace}
\reserveblockspace{remark}{\supportingblockneedspace}
\reserveblockspace{caution}{\supportingblockneedspace}
\reserveblockspace{strategy}{\supportingblockneedspace}

% ---------------------------------------------------------------------------
% Solution environment.
% Visually distinct from proof, per CONTENT_README.md v3.0 (Environment Set --
% example and solution): bold "Solution." label, upright body text, and a
% filled QED box at the end (proof uses italic "Proof." with an open box).
% Optional argument overrides the label, e.g. \begin{solution}[Alternative solution].
% Style policy: keep "Solution." inline when the body opens with prose; if the
% first real content is a block such as enumerate/itemize/display math, start
% the body on the next line with \solutionbreak instead of forcing every
% solution onto a new line.
% ---------------------------------------------------------------------------
\newenvironment{solution}[1][Solution]{%
  \par\blockneedspace{\solutionblockneedspace}%
  \pushQED{\qed}%
  \renewcommand{\qedsymbol}{\ensuremath{\blacksquare}}%
  \medskip\noindent\textbf{#1.}\enspace\ignorespaces
}{%
  \popQED\par\medskip
}
\newcommand{\solutionbreak}{\unskip\par\nobreak\smallskip\ignorespaces}

\reserveblockspace{proof}{\proofblockneedspace}

% ---------------------------------------------------------------------------
% workedexample: semantic wrapper for an example + solution pair.
%
% Needspace alone cannot keep a short example together with its solution: it
% protects only the beginning of an environment, so a short example can fit
% at a page bottom while its solution gets pushed to the next page.
%
% This environment captures its body into a vbox once, measures the combined
% height, reserves that much space, then unvboxes so the content stays
% breakable if it exceeds one page. While the body is being collected, the
% example/solution hooks must not run Needspace: those checks insert break
% penalties and stretch that can make the measured box look taller than the
% actual content, or even leak an unwanted split when the box is unboxed.
% Because the body is first boxed for measurement, keep page-anchored commands
% such as \footnote, \marginpar, and manual \hypertarget outside this wrapper.
% Implementation note: this relies on \NewEnviron's one-shot capture of \BODY.
% The inner example/solution environments are consumed during measurement and
% then replayed only via the stored box; if this wrapper is later rewritten as
% a normal \newenvironment, the pagination assumptions here will no longer hold.
% ---------------------------------------------------------------------------
\newbox\workedexamplebox
\newlength\workedexampleheight
% Cap: for worked examples taller than this, reserve only this much. Without
% the cap, a near-page-height unit forces a break on almost any page and
% leaves a near-empty trailing half-page behind. With the cap, such units
% are allowed to split naturally; short/medium units (the common case) are
% still protected in full because their measured height stays under the cap.
\newcommand{\workedexamplemaxreserve}{16\baselineskip}

\NewEnviron{workedexample}{%
  \ifmeasuringworkedexample
    \PackageError{workedexample}{Nested workedexample not allowed}{}%
  \fi
  \par
  % Measure the pair without nested Needspace interference; the outer wrapper
  % will do the single real reservation pass after the height is known.
  \measuringworkedexampletrue
  \setbox\workedexamplebox=\vbox{\hsize=\linewidth\BODY}%
  \measuringworkedexamplefalse
  \setlength\workedexampleheight{\dimexpr\ht\workedexamplebox+\dp\workedexamplebox\relax}%
  \ifdim\workedexampleheight>\workedexamplemaxreserve
    \setlength\workedexampleheight{\workedexamplemaxreserve}%
  \fi
  % Reserve the measured combined height on the actual page, then typeset the
  % body normally so overlong worked examples can still split if needed.
  \Needspace{\workedexampleheight}%
  \unvbox\workedexamplebox
  \par
}

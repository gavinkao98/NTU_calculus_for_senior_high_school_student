% Shared authoring templates for short related formulas.
% Authoring policy: within one local mathematical unit, pick the helper that
% matches the semantic pattern and avoid mixing multiple horizontal anchoring
% schemes (centered displays, aligned blocks, prose-side notes) for the same
% immediate step unless a genuinely new idea begins.
%
% Implementation note: a naive \newenvironment that wraps \[...\] introduces
% an extra TeX group around amsmath's display. amsmath's internal "no
% \parindent on the next paragraph" hook (LaTeX's \@doendpe: sets \@endpe,
% redefines \par, and rewrites \everypar so the first paragraph after the
% display eats its own indent box) only survives amsmath's own display group,
% not a second outer group: when that second group closes, \everypar and \par
% are rolled back and continuation prose is spuriously indented by 1.5em.
%
% \newdisplayenv is the house helper for defining display-math-wrapping
% environments. It uses etoolbox's \AfterEndEnvironment to install \@doendpe
% *after* \end{env} completes (after the implicit \endgroup), which reinstalls
% LaTeX's own "eat the next paragraph indent box" hook in the outer scope.
% This is the kernel-level no-indent mechanism list/item and display code
% ultimately rely on, so it is more direct than piggybacking on heading logic.
%
% Every new environment that wraps \[...\] should be declared with
% \newdisplayenv rather than \newenvironment, so this fix applies globally
% by default.
\makeatletter
\newcommand{\newdisplayenv}[3]{%
  \newenvironment{#1}{\[#2}{#3\]}%
  \AfterEndEnvironment{#1}{\@doendpe}%
}
\makeatother

\newdisplayenv{aligneddisplay}{\begin{aligned}}{\end{aligned}}
\newdisplayenv{conditiondisplay}{\begin{alignedat}{2}}{\end{alignedat}}
\newlength{\pairdisplaysep}
\newlength{\pairdisplaymaxwidth}
\setlength{\pairdisplaysep}{3em}
\setlength{\pairdisplaymaxwidth}{0.45\linewidth}
\newsavebox{\pairdisplayleftbox}
\newsavebox{\pairdisplayrightbox}
\newcommand{\pairdisplay}[2]{%
  \sbox{\pairdisplayleftbox}{$\displaystyle #1$}%
  \sbox{\pairdisplayrightbox}{$\displaystyle #2$}%
  % Side-by-side layout is reserved for two genuinely short formulas. If either
  % side grows past about 45% of the text width, stack the formulas instead.
  \ifdim\wd\pairdisplayleftbox>\pairdisplaymaxwidth
    \begin{aligneddisplay}
    #1\\
    #2
    \end{aligneddisplay}
  \else\ifdim\wd\pairdisplayrightbox>\pairdisplaymaxwidth
    \begin{aligneddisplay}
    #1\\
    #2
    \end{aligneddisplay}
  \else
    \[
    \begin{array}{@{}c@{\hspace{\pairdisplaysep}}c@{}}
      #1 & #2
    \end{array}
    \]
  \fi\fi
}

% DEPRECATED (v3.0): \iffwithconditions and \iffstackeddisplay are retained
% here only so existing chapter content continues to compile during the
% transition. The v3.0 approved-display-helper set has 5 modes (inline,
% display, aligneddisplay, conditiondisplay, pairdisplay); equivalences
% between statements are now written as ordinary display math with
% \Longleftrightarrow plus an inline or prose condition. New chapter content
% MUST NOT use the two macros below. Once every existing use has been
% migrated, both macros and this block will be removed.
% ---------------------------------------------------------------------------
\newcommand{\iffwithconditions}[3]{%
  \[
  #1
  \quad \Longleftrightarrow \quad
  \begin{cases}
    #2\\
    #3
  \end{cases}
  \]
}
\newcommand{\iffstackeddisplay}[3]{%
  \[
  #1
  \quad \Longleftrightarrow \quad
  \begin{aligned}[c]
    #2\\
    #3
  \end{aligned}
  \]
}
